/**
 * Migração: preencher patientType em appointments já importados
 *
 * Percorre todos os Appointments com patient vinculado e patientType null,
 * classifica cada um como 'novo', 'retorno' ou 'recorrente' com base no
 * histórico real do paciente, e salva.
 *
 * Uso:
 *   node scripts/migrate-patient-type.js           # executa de verdade
 *   node scripts/migrate-patient-type.js --dry-run # só simula, não salva
 */

import mongoose from 'mongoose';
import dotenv from 'dotenv';
import Appointment from '../models/Appointment.js';

dotenv.config();

const DRY_RUN = process.argv.includes('--dry-run');
const MONTHS_INACTIVE_THRESHOLD = 6;
const BATCH_SIZE = 200;

const REAL_STATUSES = ['scheduled', 'confirmed', 'paid', 'missed'];

async function classifyPatient(patientId, appointmentId, appointmentCreatedAt) {
  // Busca o appointment real mais recente desse paciente ANTES deste appointment
  const lastReal = await Appointment.findOne({
    patient: patientId,
    operationalStatus: { $in: REAL_STATUSES },
    _id: { $ne: appointmentId },
    createdAt: { $lt: appointmentCreatedAt }
  }).sort({ createdAt: -1 }).select('createdAt date').lean();

  if (!lastReal) {
    return { patientType: 'novo', patientTypeMonthsInactive: null };
  }

  const lastDate = new Date(lastReal.date || lastReal.createdAt);
  const refDate = new Date(appointmentCreatedAt);
  const diffMonths =
    (refDate.getFullYear() - lastDate.getFullYear()) * 12 +
    (refDate.getMonth() - lastDate.getMonth());

  return {
    patientType: diffMonths >= MONTHS_INACTIVE_THRESHOLD ? 'retorno' : 'recorrente',
    patientTypeMonthsInactive: diffMonths
  };
}

async function run() {
  await mongoose.connect(process.env.MONGODB_URI);
  console.log('Conectado ao MongoDB.');
  console.log(`Modo: ${DRY_RUN ? 'DRY RUN (sem salvar)' : 'REAL'}`);

  const total = await Appointment.countDocuments({
    patient: { $exists: true, $ne: null },
    operationalStatus: { $in: REAL_STATUSES },
    patientType: null
  });

  console.log(`Total de appointments a classificar: ${total}`);

  let processed = 0;
  let novo = 0;
  let retorno = 0;
  let recorrente = 0;
  let erros = 0;

  let skip = 0;
  while (skip < total) {
    const batch = await Appointment.find({
      patient: { $exists: true, $ne: null },
      operationalStatus: { $in: REAL_STATUSES },
      patientType: null
    })
      .sort({ createdAt: 1 })
      .skip(skip)
      .limit(BATCH_SIZE)
      .select('_id patient createdAt')
      .lean();

    if (batch.length === 0) break;

    const ops = [];
    for (const appt of batch) {
      try {
        const { patientType, patientTypeMonthsInactive } = await classifyPatient(
          appt.patient,
          appt._id,
          appt.createdAt
        );

        if (patientType === 'novo') novo++;
        else if (patientType === 'retorno') retorno++;
        else recorrente++;

        if (!DRY_RUN) {
          ops.push({
            updateOne: {
              filter: { _id: appt._id },
              update: { $set: { patientType, patientTypeMonthsInactive } }
            }
          });
        }
      } catch (err) {
        erros++;
        console.error(`Erro no appointment ${appt._id}:`, err.message);
      }
    }

    if (!DRY_RUN && ops.length > 0) {
      await Appointment.bulkWrite(ops);
    }

    processed += batch.length;
    skip += BATCH_SIZE;
    console.log(`  Processados: ${processed}/${total} (novo=${novo} retorno=${retorno} recorrente=${recorrente} erros=${erros})`);
  }

  console.log('\nMigracao concluida:');
  console.log(`  Novos:       ${novo}`);
  console.log(`  Retornos:    ${retorno}`);
  console.log(`  Recorrentes: ${recorrente}`);
  console.log(`  Erros:       ${erros}`);
  if (DRY_RUN) console.log('  (dry-run: nenhuma alteracao salva)');

  await mongoose.disconnect();
}

run().catch(err => {
  console.error('Falha fatal:', err);
  process.exit(1);
});
