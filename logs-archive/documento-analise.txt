🔴 FASE 3 — CONTEXT SERVICE (SEU MAIOR GANHO)
export async function loadContext(userId) {
  const raw = await redis.get(`ctx:${userId}`);

  if (!raw) {
    return {
      history: [],
      stage: "NEW",
      collectedData: {}
    };
  }

  return JSON.parse(raw);
}

export async function saveContext(userId, state) {
  await redis.set(
    `ctx:${userId}`,
    JSON.stringify(state),
    "EX",
    60 * 60 * 24 * 7
  );
}
