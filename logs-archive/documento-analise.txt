temas.forEach(async ({ tema, subTema }) => {

      preset: "explosao_viral",
      especialidadeId: "fonoaudiologia",
      modo: "veo",
      tema,
      subTema,
      usarScriptExemplo: false  // Gera roteiro novo para cada tema
    })
  });
});
