# 🔥 1. APPOINTMENT V2 ENGINE (PRINCIPAL)

```js
import Appointment from "../../models/Appointment.js";
import Patient from "../../models/Patient.js";
import Doctor from "../../models/Doctor.js";
import Session from "../../models/Session.js";
import Payment from "../../models/Payment.js";

import PatientService from "../patient.service.js";
import { findDoctorByName } from "../../utils/doctorResolver.js";
import { normalizeSessionType } from "../../utils/sessionTypeResolver.js";

class AppointmentV2Engine {

  /**
   * ENTRADA ÚNICA DO SISTEMA
   * qualquer fluxo (agenda externa, crm, webhook) passa aqui
   */
  async executeBooking(input) {
    const {
      source = "unknown",
      professionalName,
      doctorId,
      patientId,
      patientInfo,
      date,
      time,
      specialty,
      serviceType = "evaluation",
      sessionValue = 0,
      paymentMethod = "pix",
      billingType = "particular",
      notes = ""
    } = input;

    // 1. VALIDAR INPUT BÁSICO
    if (!date || !time) {
      throw new Error("date e time são obrigatórios");
    }

    // 2. RESOLVER DOCTOR
    let doctor = null;

    if (doctorId) {
      doctor = await Doctor.findById(doctorId);
    }

    if (!doctor && professionalName) {
      doctor = await findDoctorByName(professionalName);
    }

    if (!doctor) {
      throw new Error("Doutor não encontrado");
    }

    // 3. RESOLVER PACIENTE (IDEMPOTENTE)
    const patient = await PatientService.resolve({
      patientId,
      patientInfo
    });

    // 4. VERIFICAR DUPLICIDADE (CRÍTICO)
    const existing = await Appointment.findOne({
      doctor: doctor._id,
      date,
      time,
      operationalStatus: { $nin: ["canceled", "cancelado", "cancelada"] }
    }).lean();

    if (existing) {
      return {
        success: true,
        skipped: true,
        reason: "slot_already_taken",
        appointmentId: existing._id
      };
    }

    // 5. CRIAR APPOINTMENT (SINGLE SOURCE OF TRUTH)
    const appointment = await Appointment.create({
      patient: patient._id,
      doctor: doctor._id,
      patientInfo: patientInfo || {},
      professionalName: doctor.fullName,
      date,
      time,
      specialty: normalizeSessionType(specialty || doctor.specialty),
      serviceType,
      sessionValue,
      paymentMethod,
      billingType,
      notes,
      operationalStatus: "scheduled",
      clinicalStatus: "pending",
      paymentStatus: "pending",
      visualFlag: "pending",
      metadata: {
        origin: { source }
      }
    });

    // 6. CRIAR SESSION (DERIVADO)
    const session = await Session.create({
      patient: patient._id,
      doctor: doctor._id,
      appointmentId: appointment._id,
      sessionType: normalizeSessionType(specialty || doctor.specialty),
      date,
      time,
      sessionValue,
      status: "scheduled",
      paymentStatus: "pending",
      visualFlag: "pending",
      notes
    });

    // 7. CRIAR PAYMENT (DERIVADO)
    const payment = await Payment.create({
      patient: patient._id,
      session: session._id,
      appointment: appointment._id,
      serviceType,
      amount: sessionValue,
      paymentMethod,
      billingType,
      status: "pending",
      paymentDate: date,
      source: "appointment_v2"
    });

    // 8. LINK FINAL
    session.paymentId = payment._id;
    await session.save();

    appointment.session = session._id;
    appointment.payment = payment._id;
    await appointment.save();

    return {
      success: true,
      appointmentId: appointment._id,
      sessionId: session._id,
      paymentId: payment._id,
      patientId: patient._id
    };
  }

  /**
   * CONFIRMAÇÃO (PROMOTION FLOW)
   */
  async confirmAppointment(appointmentId, extra = {}) {
    const appointment = await Appointment.findById(appointmentId);

    if (!appointment) {
      throw new Error("Appointment não encontrado");
    }

    if (appointment.operationalStatus !== "pre_agendado" &&
        appointment.operationalStatus !== "scheduled") {
      return {
        success: true,
        skipped: true,
        reason: "invalid_status"
      };
    }

    appointment.operationalStatus = "scheduled";
    appointment.clinicalStatus = "confirmed";
    appointment.updatedAt = new Date();

    await appointment.save();

    return {
      success: true,
      appointmentId
    };
  }

  /**
   * CANCELAMENTO CENTRALIZADO
   */
  async cancelAppointment(appointmentId, reason = "") {
    const appointment = await Appointment.findById(appointmentId);

    if (!appointment) {
      throw new Error("Appointment não encontrado");
    }

    appointment.operationalStatus = "canceled";
    appointment.paymentStatus = "canceled";
    appointment.clinicalStatus = "canceled";
    appointment.canceledReason = reason;
    appointment.canceledAt = new Date();

    await appointment.save();

    return { success: true };
  }
}

export default new AppointmentV2Engine();
```
