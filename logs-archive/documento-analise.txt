```javascript
// 🔹 AmandaOrchestrator.js - snippet Regra 5 + feriados

const { Appointment } = require("../models/Appointment");
const nationalHolidaysBR = require("../config/feriadosBR"); // array de datas YYYY-MM-DD

async function filterUnavailableSlots(doctor, allSlots) {
  // Pega todos slots ocupados por pacotes contínuos
  const occupiedSlots = await Appointment.find({
    doctorId: doctor._id,
    packageId: { $exists: true },        // apenas pacotes/sessões contínuas
    status: { $in: ["scheduled", "done"] } 
  });

  // Filtra slots que estão em feriado ou já ocupados
  const filteredSlots = allSlots.filter(slot => {
    const slotDateStr = slot.date.toISOString().split("T")[0];

    // 🔹 Não oferecer se for feriado
    if (nationalHolidaysBR.includes(slotDateStr)) return false;

    // 🔹 Não oferecer se algum paciente contínuo já ocupa
    const isOccupied = occupiedSlots.some(occ => 
      occ.dayOfWeek === slot.date.getDay() &&
      occ.hour === slot.date.getHours()
    );

    return !isOccupied;
  });

  return filteredSlots;
}

// 🔹 Uso dentro de handleTriagemResponse() ou fluxo ALTA_INTENCAO
async function getAvailableSlotsForAmanda(doctor, allSlots, context) {
  const slots = await filterUnavailableSlots(doctor, allSlots);

  // Se não tiver slots, prepara resposta rápida
  if (!slots.length) {
    context.offerSlotsImmediately = false;
    return "Não tenho horários disponíveis neste período. Posso te oferecer outras datas?";
  }

  context.offerSlotsImmediately = true;
  return slots.map(s => `${s.date.toLocaleDateString()} às ${s.date.getHours()}:00`).slice(0,5);
}

module.exports = { filterUnavailableSlots, getAvailableSlotsForAmanda };
```

---

### **💡 Detalhes importantes**

1. `Appointment.packageId` → indica que a sessão faz parte de um pacote contínuo.
2. `status: ["scheduled","done"]` → só considera sessões agendadas ou já realizadas. Canceladas não bloqueiam.
3. `nationalHolidaysBR` → array de feriados nacionais em formato `YYYY-MM-DD`. Você pode manter no `config/feriadosBR.js`.
4. O retorno do `getAvailableSlotsForAmanda` já está pronto para usar no **handler ALTA_INTENCAO**.
