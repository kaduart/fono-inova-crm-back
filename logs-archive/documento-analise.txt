Bash Run audit filtering only completed appointments
IN
cd /home/user/projetos/crm/back && cat > auditoria_atendidos.mjs << 'SCRIPT'
import mongoose from 'mongoose';
import dotenv from 'dotenv';
dotenv.config();

await mongoose.connect(process.env.MONGO_URI);
const db = mongoose.connection.db;

const fmt = d => d ? new Date(d).toISOString().slice(0,10).split('-').reverse().join('/') : '—';
const mesAno = d => { if (!d) return '??'; const x = new Date(d); return `${String(x.getUTCMonth()+1).padStart(2,'0')}/${x.getUTCFullYear()}`; };

// ── 1. PACOTES ANTIGOS (type=convenio) ─────────────────────────────────
// Busca sessions vinculadas a cada package e filtra só completed
const packages = await db.collection('packages').find({
  type: 'convenio',
  provider: { $ne: null }
}).sort({ date: 1 }).toArray();

const pkgIds = packages.map(p => p._id);
const patientIds = [...new Set(packages.map(p => String(p.patient)).filter(Boolean))];

// Busca patients
const patients = await db.collection('patients').find({
  _id: { $in: patientIds.map(id => new mongoose.Types.ObjectId(id)) }
}).toArray();
const patMap = {};
for (const p of patients) patMap[String(p._id)] = p.name || p.fullName || '??';

// Busca appointments de convenio do período jan-mar/2026 com status completed
// vinculados a packages
const pkgAppts = await db.collection('appointments').find({
  packageId: { $in: pkgIds },
  operationalStatus: 'completed',
  date: { $gte: new Date('2026-01-01') }
}).sort({ date: 1 }).toArray();

console.log('\n═══════════════════════════════════════════════════════════════════');
console.log('  ATENDIMENTOS CONVÊNIO — SÓ COMPLETED — Jan–Jun 2026');
console.log('═══════════════════════════════════════════════════════════════════\n');

// Agrupa por paciente→mês
const porMesAppt = {};
for (const a of pkgAppts) {
  const mes = mesAno(a.date);
  const pkg = packages.find(p => String(p._id) === String(a.packageId));
  const provider = pkg?.provider || pkg?.insuranceProvider || 'convenio';
  const patName = patMap[String(a.patient)] || a.patientInfo?.fullName || '??';
  const key = `${mes}|${patName}`;
  if (!porMesAppt[key]) porMesAppt[key] = { mes, patName, provider, sessoes: [] };
  porMesAppt[key].sessoes.push({ data: fmt(a.date), specialty: a.specialty, status: a.operationalStatus });
}

// Também tenta via sessions (caso appointments não tenham packageId)
// Busca sessions linked to packages
const pkgSessions = await db.collection('sessions').find({
  packageId: { $in: pkgIds },
  status: 'completed',
  date: { $gte: new Date('2026-01-01') }
}).sort({ date: 1 }).toArray();

const sessAppointmentIds = pkgSessions.map(s => s.appointment).filter(Boolean);
const sessAppts = await db.collection('appointments').find({
  _id: { $in: sessAppointmentIds },
  operationalStatus: 'completed'
}).toArray();
const sessApptMap = {};
for (const a of sessAppts) sessApptMap[String(a._id)] = a;

for (const s of pkgSessions) {
  const appt = sessApptMap[String(s.appointment)];
  const pkg = packages.find(p => String(p._id) === String(s.packageId));
  const provider = pkg?.provider || pkg?.insuranceProvider || 'convenio';
  const patId = s.patient || appt?.patient || pkg?.patient;
  const patName = patMap[String(patId)] || appt?.patientInfo?.fullName || '??';
  const mes = mesAno(s.date || appt?.date);
  const key = `${mes}|${patName}`;
  if (!porMesAppt[key]) porMesAppt[key] = { mes, patName, provider, sessoes: [] };
  // evita duplicata se já veio via appointment
  const dataStr = fmt(s.date || appt?.date);
  const specialty = appt?.specialty || s.specialty || '—';
  if (!porMesAppt[key].sessoes.find(x => x.data === dataStr && x.specialty === specialty)) {
    porMesAppt[key].sessoes.push({ data: dataStr, specialty, status: 'completed' });
  }
}

// ── 2. APPOINTMENTS NOVOS (paymentMode=convenio, sem package) ─────────
const newAppts = await db.collection('appointments').find({
  paymentMode: 'convenio',
  operationalStatus: 'completed',
  date: { $gte: new Date('2026-01-01') }
}).sort({ date: 1 }).toArray();

for (const a of newAppts) {
  const mes = mesAno(a.date);
  const patName = patMap[String(a.patient)] || a.patientInfo?.fullName || '??';
  const provider = a.insuranceProvider || 'convenio';
  const key = `${mes}|${patName}`;
  if (!porMesAppt[key]) porMesAppt[key] = { mes, patName, provider, sessoes: [] };
  const dataStr = fmt(a.date);
  if (!porMesAppt[key].sessoes.find(x => x.data === dataStr && x.specialty === a.specialty)) {
    porMesAppt[key].sessoes.push({ data: dataStr, specialty: a.specialty, status: 'completed' });
  }
}

// ── Output agrupado por mês ────────────────────────────────────────────
const mesList = [...new Set(Object.values(porMesAppt).map(x => x.mes))].sort((a,b) => {
  const [ma,ya] = a.split('/').map(Number);
  const [mb,yb] = b.split('/').map(Number);
  return ya !== yb ? ya-yb : ma-mb;
});

let grandTotal = 0;
for (const mes of mesList) {
  const grupos = Object.values(porMesAppt).filter(x => x.mes === mes);
  const totalSessoes = grupos.reduce((s,g) => s + g.sessoes.length, 0);
  grandTotal += totalSessoes;
  console.log(`▓▓▓ ${mes} ─── ${totalSessoes} sessão(ões) atendidas ─── ${grupos.length} paciente(s) ▓▓▓`);
  for (const g of grupos.sort((a,b) => a.patName.localeCompare(b.patName))) {
    console.log(`  👤 ${g.patName}  [${g.provider}]  — ${g.sessoes.length} sessão(ões)`);
    for (const s of g.sessoes.sort((a,b) => a.data.localeCompare(b.data))) {
      console.log(`     • ${s.data}  ${s.specialty}`);
    }
  }
  console.log();
}

console.log(`═══════════════════════════════════════════════════════════════════`);
console.log(`  TOTAL GERAL: ${grandTotal} sessões atendidas (completed)`);
console.log(`═══════════════════════════════════════════════════════════════════\n`);

await mongoose.disconnect();

Bash Query completed appointments linked to convenio packages
IN
MONGO_URI="$(grep MONGO_URI .env | cut -d'"' -f2)" node --input-type=module << 'EOF'
import mongoose from 'mongoose';
import dotenv from 'dotenv';
dotenv.config();
await mongoose.connect(process.env.MONGO_URI);
const db = mongoose.connection.db;

const fmt = d => { if (!d) return '—'; const x = new Date(d); return `${String(x.getUTCDate()).padStart(2,'0')}/${String(x.getUTCMonth()+1).padStart(2,'0')}/${x.getUTCFullYear()}`; };
const mesAno = d => { if (!d) return '??'; const x = new Date(d); return `${String(x.getUTCMonth()+1).padStart(2,'0')}/${x.getUTCFullYear()}`; };

// Busca patients para resolver nomes
const allPatients = await db.collection('patients').find({}).project({ name: 1, fullName: 1 }).toArray();
const patMap = {};
for (const p of allPatients) patMap[String(p._id)] = p.name || p.fullName || '??';

// Busca packages de convenio (todas as providers)
const packages = await db.collection('packages').find({ type: 'convenio' }).toArray();
const pkgMap = {};
for (const p of packages) pkgMap[String(p._id)] = p;
const pkgIds = packages.map(p => p._id);

console.log(`Packages convenio encontrados: ${packages.length}`);
// Ver provedores
const providers = [...new Set(packages.map(p => p.insuranceProvider || p.provider || '??'))];
console.log('Providers:', providers.join(', '));

// Busca TODOS appointments completed vinculados a packages de convenio
const appts = await db.collection('appointments').find({
  $or: [
    { package: { $in: pkgIds } },
    { packageId: { $in: pkgIds } }
  ],
  operationalStatus: 'completed',
  date: { $gte: new Date('2026-01-01'), $lte: new Date('2026-12-31') }
}).sort({ date: 1 }).toArray();

console.log(`Appointments completed encontrados: ${appts.length}`);

// Agrupa por mês → provider → paciente
const result = {};
for (const a of appts) {
  const pkgId = String(a.package || a.packageId);
  const pkg = pkgMap[pkgId];
  const provider = pkg?.insuranceProvider || pkg?.provider || 'convenio';
  const mes = mesAno(a.date);
  const patName = patMap[String(a.patient)] || a.patientInfo?.fullName || a.patientName || '??';
  const key = `${mes}||${provider}||${patName}`;
  if (!result[key]) result[key] = { mes, provider, patName, sessoes: [] };
  result[key].sessoes.push({ data: fmt(a.date), specialty: a.specialty || '—' });
}

// Sort e imprime
const meses = [...new Set(Object.values(result).map(x => x.mes))].sort((a,b) => {
  const [ma,ya] = a.split('/').map(Number);
  const [mb,yb] = b.split('/').map(Number);
  return ya !== yb ? ya-yb : ma-mb;
});

let grandTotal = 0;
console.log('\n═══════════════════════════════════════════════════════');
console.log('  ATENDIMENTOS CONVÊNIO — SÓ COMPLETED — Jan–Jun 2026');
console.log('═══════════════════════════════════════════════════════\n');

for (const mes of meses) {
  const grupos = Object.values(result).filter(x => x.mes === mes);
  const total = grupos.reduce((s,g) => s + g.sessoes.length, 0);
  grandTotal += total;
  console.log(`▓▓▓ ${mes} — ${total} sessão(ões) — ${grupos.length} paciente(s) ▓▓▓`);
  // Agrupa por provider dentro do mês
  const providers = [...new Set(grupos.map(g => g.provider))];
  for (const prov of providers) {
    const gs = grupos.filter(g => g.provider === prov);
    if (providers.length > 1) console.log(`  [${prov}]`);
    for (const g of gs.sort((a,b) => a.patName.localeCompare(b.patName))) {
      console.log(`  👤 ${g.patName} — ${g.sessoes.length} sessão(ões)`);
      for (const s of g.sessoes) console.log(`     • ${s.data}  ${s.specialty}`);
    }
  }
  console.log();
}

console.log(`═══════════════════════════════════════════════════════`);
console.log(`  TOTAL GERAL: ${grandTotal} sessões atendidas (completed)`);
console.log(`═══════════════════════════════════════════════════════\n`);

await mongoose.disconnect();
