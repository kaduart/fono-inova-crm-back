# ⚙️ 2. SERVICE PRINCIPAL

## `generateSuggestedSlots.js`

```js id="svc1"
const Appointment = require('../models/Appointment');
const Doctor = require('../models/Doctor');
const dayjs = require('dayjs');

async function generateSuggestedSlots({
  doctorId,
  patientId,
  dateFrom,
  dateTo
}) {
  const doctor = await Doctor.findById(doctorId);
  if (!doctor) throw new Error('Doctor not found');

  const start = dateFrom ? dayjs(dateFrom) : dayjs();
  const end = dateTo ? dayjs(dateTo).add(7, 'day') : dayjs().add(7, 'day');

  const appointments = await Appointment.find({
    doctor: doctorId,
    date: { $gte: start.toDate(), $lte: end.toDate() },
    operationalStatus: { $ne: 'canceled' }
  });

  const occupied = new Set(
    appointments.map(a => `${a.date.toISOString().slice(0,10)}_${a.time}`)
  );

  const slots = [];

  // horários padrão da clínica (pode vir do doctor depois)
  const workingHours = [
    "08:00","09:00","10:00","11:00",
    "14:00","15:00","16:00","17:00"
  ];

  let cursor = start.clone();

  while (cursor.isBefore(end)) {
    const dateStr = cursor.format('YYYY-MM-DD');

    for (const time of workingHours) {
      const key = `${dateStr}_${time}`;

      if (occupied.has(key)) continue;

      const score = calculateScore({
        doctor,
        patientId,
        date: dateStr,
        time
      });

      slots.push({
        date: dateStr,
        time,
        doctorId,
        score,
        reason: score >= 80
          ? "Alta prioridade (slot ideal)"
          : "Slot alternativo disponível"
      });
    }

    cursor = cursor.add(1, 'day');
  }

  return slots
    .sort((a, b) => b.score - a.score)
    .slice(0, 5);
}

module.exports = generateSuggestedSlots;
```

# 🧠 3. MOTOR DE SCORE (SIMPLIFICADO V1)

```js id="score1"
function calculateScore({ doctor, patientId, date, time }) {
  let score = 70; // base: slot livre

  // horário preferido do médico
  if (doctor.preferredHours?.includes(time)) {
    score += 20;
  }

  // bônus simples (pode evoluir depois)
  const hour = parseInt(time.split(':')[0]);

  if (hour >= 9 && hour <= 11) score += 10;   // manhã boa adesão
  if (hour >= 14 && hour <= 16) score += 5;   // tarde ok

  return Math.min(score, 100);
}
```

# 📦 4. CONTROLLER

```js id="ctrl1"
const generateSuggestedSlots = require('../services/generateSuggestedSlots');

async function agendaSuggestions(req, res) {
  try {
    const { doctorId, patientId, dateFrom, dateTo } = req.body;

    const suggestions = await generateSuggestedSlots({
      doctorId,
      patientId,
      dateFrom,
      dateTo
    });

    return res.json({
      success: true,
      data: suggestions
    });

  } catch (err) {
    return res.status(500).json({
      success: false,
      message: err.message
    });
  }
}

module.exports = { agendaSuggestions };
```
