```js
// scripts/fix-sessionValue-from-source.js

const DRY_RUN = process.env.DRY_RUN !== 'false';

for (const session of sessions) {
  let newValue = null;

  const appointment = await Appointment.findById(session.appointmentId);
  const payment = await Payment.findOne({ sessionId: session._id });
  
  // 1. PACKAGE (PRIORIDADE MÁXIMA)
  if (appointment?.packageId) {
    const pkg = await Package.findById(appointment.packageId);
    if (pkg?.sessionValue > 0) {
      newValue = pkg.sessionValue;
    }
  }

  // 2. PAYMENT (VALOR REAL)
  if (!newValue && payment?.amount > 0) {
    newValue = payment.amount;
  }

  // 3. APPOINTMENT (LEGADO)
  if (!newValue && appointment?.sessionValue > 1) {
    newValue = appointment.sessionValue;
  }

  // 4. FALLBACK CONTROLADO
  if (!newValue) {
    if (appointment?.service === 'evaluation') newValue = 200;
    else newValue = 150;
  }

  // ignora lixo
  if (newValue <= 0.1) continue;

  if (!DRY_RUN) {
    await Session.updateOne(
      { _id: session._id },
      {
        $set: {
          value: newValue,
          valueSource: appointment?.packageId
            ? 'package'
            : payment?.amount
            ? 'payment'
            : appointment?.sessionValue
            ? 'appointment'
            : 'fallback'
        }
      }
    );
```

```js
// services/dailyClosingV2.js

async function getDailyClosing(date) {
  const start = startOfDay(date);
  const end = endOfDay(date);

  // 🟢 SESSÕES (fonte de verdade)
  const sessions = await Session.find({
    date: { $gte: start, $lte: end }
  });

  // 💰 PAGAMENTOS
  const payments = await Payment.find({
    date: { $gte: start, $lte: end },
    status: 'paid'
  });

  // ========================
  // 📊 PRODUÇÃO (REAL)
  // ========================
  const completedSessions = sessions.filter(
    s => s.status === 'completed'
  );

  const productionValue = completedSessions.reduce(
    (sum, s) => sum + (s.value || 0),
    0
  );

  // ========================
  // 📊 PREVISÃO (AGENDA)
  // ========================
  const futureSessions = sessions.filter(
    s => ['scheduled', 'pending'].includes(s.status)
  );

  const expectedValue = futureSessions.reduce(
    (sum, s) => sum + (s.value || 0),
    0
  );

  // ========================
  // 💰 RECEBIDO
  // ========================
  const receivedValue = payments.reduce(
    (sum, p) => sum + (p.amount || 0),
    0
  );

  // ========================
  // 💳 POR MÉTODO
  // ========================
  const byMethod = {
    dinheiro: 0,
    pix: 0,
    cartao: 0
  };

  for (const p of payments) {
    if (p.method === 'pix') byMethod.pix += p.amount;
    else if (p.method === 'dinheiro') byMethod.dinheiro += p.amount;
    else byMethod.cartao += p.amount;
  }

  return {
    summary: {
      production: productionValue,
      expected: expectedValue,
      received: receivedValue
    },
    byMethod
  };
}
```
