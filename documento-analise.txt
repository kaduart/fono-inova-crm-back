// meta-post.js
const PAGE_ID = process.env.META_PAGE_ID;
const IG_USER_ID = process.env.META_IG_USER_ID;
const ACCESS_TOKEN = process.env.META_ACCESS_TOKEN; // token de página (não expira se for do sistema)

// Post de FOTO no Instagram
async function postarFotoInstagram(imageUrl, legenda) {
  // 1. Cria container
  const container = await fetch(
    `https://graph.facebook.com/v19.0/${IG_USER_ID}/media`, {
    method: 'POST',
    body: new URLSearchParams({
      image_url: imageUrl,  // URL pública da imagem
      caption: legenda,
      access_token: ACCESS_TOKEN
    })
  }).then(r => r.json());

  // 2. Publica
  const post = await fetch(
    `https://graph.facebook.com/v19.0/${IG_USER_ID}/media_publish`, {
    method: 'POST',
    body: new URLSearchParams({
      creation_id: container.id,
      access_token: ACCESS_TOKEN
    })
  }).then(r => r.json());

  console.log('✅ Instagram postado:', post.id);
  return post;
}

// Post de VÍDEO (Reels) no Instagram
async function postarReelInstagram(videoUrl, legenda) {
  // 1. Cria container do reel
  const container = await fetch(
    `https://graph.facebook.com/v19.0/${IG_USER_ID}/media`, {
    method: 'POST',
    body: new URLSearchParams({
      media_type: 'REELS',
      video_url: videoUrl,   // URL pública do .mp4
      caption: legenda,
      share_to_feed: 'true',
      access_token: ACCESS_TOKEN
    })
  }).then(r => r.json());

  // 2. Aguarda processamento (vídeo demora ~30s)
  await new Promise(r => setTimeout(r, 35000));

  // 3. Publica
  const post = await fetch(
    `https://graph.facebook.com/v19.0/${IG_USER_ID}/media_publish`, {
    method: 'POST',
    body: new URLSearchParams({
      creation_id: container.id,
      access_token: ACCESS_TOKEN
    })
  }).then(r => r.json());

  console.log('✅ Reel postado:', post.id);
  return post;
}

// heygen-to-instagram.js
async function gerarEPostarVideo(tema) {
  // 1. Claude gera roteiro curto
  const roteiro = await anthropic.messages.create({
    model: 'claude-sonnet-4-6',
    max_tokens: 200,
    messages: [{
      role: 'user',
      content: `Roteiro de 30s para vídeo do Instagram da Clínica Fono Inova sobre: "${tema}".
      Formato: fala corrida, sem marcações, tom de especialista acolhedora.
      Máximo 80 palavras.`
    }]
  });
  const script = roteiro.content[0].text;

  // 2. HeyGen gera vídeo com avatar
  const heygenRes = await fetch('https://api.heygen.com/v2/video/generate', {
    method: 'POST',
    headers: {
      'X-Api-Key': process.env.HEYGEN_API_KEY,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      video_inputs: [{
        character: {
          type: 'avatar',
          avatar_id: process.env.HEYGEN_AVATAR_ID,
          avatar_style: 'normal'
        },
        voice: {
          type: 'text',
          input_text: script,
          voice_id: process.env.HEYGEN_VOICE_ID
        }
      }],
      dimension: { width: 1080, height: 1920 }, // 9:16 Reels
      aspect_ratio: '9:16'
    })
  }).then(r => r.json());

  const videoId = heygenRes.data.video_id;
  console.log('🎬 Vídeo sendo gerado, ID:', videoId);

  // 3. Aguarda renderização (polling)
  let videoUrl = null;
  for (let i = 0; i < 20; i++) {
    await new Promise(r => setTimeout(r, 15000)); // espera 15s
    const status = await fetch(`https://api.heygen.com/v1/video_status.get?video_id=${videoId}`, {
      headers: { 'X-Api-Key': process.env.HEYGEN_API_KEY }
    }).then(r => r.json());

    if (status.data.status === 'completed') {
      videoUrl = status.data.video_url;
      break;
    }
    console.log(`⏳ Aguardando vídeo... tentativa ${i + 1}`);
  }

  if (!videoUrl) throw new Error('HeyGen timeout');

  // 4. Gera legenda com Claude
  const legendaRes = await anthropic.messages.create({
    model: 'claude-sonnet-4-6',
    max_tokens: 150,
    messages: [{
      role: 'user',
      content: `Legenda Instagram para vídeo sobre "${tema}" da Clínica Fono Inova.
      Máx 100 palavras. Inclua 5 hashtags relevantes. CTA: "Agende pelo WhatsApp 📲"`
    }]
  });
  const legenda = legendaRes.content[0].text;

  // 5. Posta no Instagram
  await postarReelInstagram(videoUrl, legenda);
  console.log('🚀 Pipeline completo!');
}

async function gerarImagemLeonardo(prompt) {
  // 1. Gera imagem
  const gen = await fetch('https://cloud.leonardo.ai/api/rest/v1/generations', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${process.env.LEONARDO_API_KEY}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      prompt,
      modelId: 'b24e16ff-06e3-43eb-8d33-4416c2d75876', // Leonardo Diffusion XL
      width: 1080,
      height: 1080,
      num_images: 1
    })
  }).then(r => r.json());

  const genId = gen.sdGenerationJob.generationId;

  // 2. Polling até ficar pronto
  for (let i = 0; i < 10; i++) {
    await new Promise(r => setTimeout(r, 5000));
    const status = await fetch(
      `https://cloud.leonardo.ai/api/rest/v1/generations/${genId}`,
      { headers: { 'Authorization': `Bearer ${process.env.LEONARDO_API_KEY}` } }
    ).then(r => r.json());

    const imgs = status.generations_by_pk?.generated_images;
    if (imgs?.length > 0) return imgs[0].url;
  }
  throw new Error('Leonardo timeout');
}
