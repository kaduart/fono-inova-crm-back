// ─── Quebra texto em linhas ─────────────────────────────────────────────────
function quebrarTexto(texto, maxChars) {
  const palavras = texto.split(' ');
  const linhas = [];
  let linha = '';

  for (const palavra of palavras) {
    if ((linha + ' ' + palavra).trim().length <= maxChars) {
      linha = (linha + ' ' + palavra).trim();
    } else {
      if (linha) linhas.push(linha);
      linha = palavra;
    }
  }
  if (linha) linhas.push(linha);
  return linhas;
}

// Extrai título e corpo do conteúdo do post
function extrairTextos(titulo, conteudo) {
  // Título: até 45 chars
  const tituloFinal = (titulo || '')
    .replace(/[<>&"']/g, c => ({ '<': '&lt;', '>': '&gt;', '&': '&amp;', '"': '&quot;', "'": '&apos;' }[c]))
    .substring(0, 55);

  // Corpo: primeiras 2 frases do conteúdo, sem o título
  const frases = conteudo
    .split(/(?<=[.!?])\s+/)
    .filter(f => f.length > 20)
    .slice(0, 3)
    .join(' ')
    .substring(0, 160);

  const corpoEscapado = frases.replace(/[<>&"']/g, c =>
    ({ '<': '&lt;', '>': '&gt;', '&': '&amp;', '"': '&quot;', "'": '&apos;' }[c]));

  return { titulo: tituloFinal, corpo: corpoEscapado };
}

/**
 * Gera imagem branded:
 * 1. Usa fotoUrl (DALL-E 3 ou outra) como fundo
 * 2. Aplica overlay colorido com opacidade
 * 3. Composita SVG de texto
 * 4. Faz upload no Cloudinary
 *
 * @param {Object} params
 * @param {string} params.fotoUrl        URL da foto gerada pela IA
 * @param {string} params.titulo         Título do post
 * @param {string} params.conteudo       Texto do post
 * @param {string} params.especialidadeId ID da especialidade
 * @param {number} [params.templateIndex] Forçar índice de template (0-4); padrão: aleatório
 * @returns {string} URL Cloudinary da imagem final branded
 */
export async function gerarImagemBranded({ fotoUrl, titulo, conteudo, especialidadeId, templateIndex }) {
  const SIZE = 1080;

  // 1. Seleciona template
  const idx = templateIndex !== undefined
    ? templateIndex % PALETA.length
    : Math.floor(Math.random() * PALETA.length);
  const paleta = PALETA[idx];

  console.log(`🎨 [BRAND] Template: ${paleta.id} | Especialidade: ${especialidadeId}`);

  // 2. Busca foto da IA como fundo
  let fotoBuf;
  try {
    const r = await fetch(fotoUrl, { signal: AbortSignal.timeout(20000) });
    fotoBuf = Buffer.from(await r.arrayBuffer());
  } catch {
    // Fundo sólido preto se foto falhar
    fotoBuf = await sharp({
      create: { width: SIZE, height: SIZE, channels: 4, background: { r: 20, g: 60, b: 40, alpha: 1 } }
    }).jpeg().toBuffer();
  }

  // 3. Redimensiona foto para 1080x1080 (crop center)
  const fotoResized = await sharp(fotoBuf)
    .resize(SIZE, SIZE, { fit: 'cover', position: 'center' })
    .jpeg({ quality: 90 })
    .toBuffer();

  // 4. Cria overlay colorido com opacidade
  const alpha = Math.round(paleta.overlayAlpha * 255);
  const overlay = await sharp({
    create: {
      width: SIZE, height: SIZE, channels: 4,
      background: { r: paleta.overlayR, g: paleta.overlayG, b: paleta.overlayB, alpha }
    }
  }).png().toBuffer();

  // 5. Composita foto + overlay
  const comBase = await sharp(fotoResized)
    .composite([{ input: overlay, blend: 'over' }])
    .jpeg({ quality: 90 })
    .toBuffer();

  // 6. Gera SVG de texto
  const label = LABELS[especialidadeId] || 'FONO INOVA';
  const { titulo: tit, corpo } = extrairTextos(titulo, conteudo);

  const svgStr = gerarSVG({ paleta, label, titulo: tit, corpo, largura: SIZE, altura: SIZE });
  const svgBuf = Buffer.from(svgStr);

  // 7. Composita texto sobre a imagem
  const imagemFinal = await sharp(comBase)
    .composite([{ input: svgBuf, blend: 'over' }])
    .jpeg({ quality: 92 })
    .toBuffer();

  // 8. Upload no Cloudinary
  const base64 = `data:image/jpeg;base64,${imagemFinal.toString('base64')}`;
  const result = await cloudinary.uploader.upload(base64, {
    folder: 'fono-inova/branded',
    public_id: `${especialidadeId}_branded_${Date.now()}`,
  });

  console.log(`✅ [BRAND] Imagem branded: ${result.secure_url}`);
  return result.secure_url;
}

/**
 * Gera apenas o overlay branded sobre uma URL de foto existente
 * (útil para regenerar sem chamar DALL-E novamente)
 */
export async function aplicarBrandingSobreFoto(fotoUrl, titulo, conteudo, especialidadeId) {
  return gerarImagemBranded({ fotoUrl, titulo, conteudo, especialidadeId });
}

function escapeSvg(s = '') {
  return String(s).replace(/[<>&"']/g, c =>
    ({ '<': '&lt;', '>': '&gt;', '&': '&amp;', '"': '&quot;', "'": '&apos;' }[c]));
}

/** Tamanho de fonte dinâmico baseado no comprimento da palavra */
function fontSizeParaLinha(linha) {
  const n = linha.length;
  if (n <= 10) return 82;
  if (n <= 13) return 70;
  if (n <= 16) return 60;
  return 52;
}

  <defs>
    <filter id="sh">

/**
 * Gera imagem branded estilo Fono Inova:
 * 1. Foto limpa (sem overlay) no topo
 * 2. Banda verde sólida no rodapé com especialidade + hook + logo
 * 3. Blobs decorativos nos cantos
 * 4. Upload Cloudinary
 *
 * @param {Object}  p
 * @param {string}  [p.fotoUrl]       URL da foto gerada pela IA (alternativa ao buffer)
 * @param {Buffer}  [p.fotoBuffer]    Buffer direto da foto (preferível, evita double-fetch)
 * @param {string}  p.titulo          Título do post (usado como hook/subtítulo)
 * @param {string}  p.conteudo        Corpo do post
 * @param {string}  p.especialidadeId ID da especialidade
 * @param {number}  [p.templateIndex] Forçar índice de template (0–4)
 * @returns {Promise<string>} URL Cloudinary da imagem branded
 */
export async function gerarImagemBranded({ fotoUrl, fotoBuffer, titulo, conteudo, especialidadeId, templateIndex }) {
  const SIZE   = 1080;
  const BAND_H = 420;          // altura da banda inferior (~39%)
  const BAND_Y = SIZE - BAND_H; // y=660

  const idx  = templateIndex !== undefined
    ? templateIndex % TEMPLATES.length
    : Math.floor(Math.random() * TEMPLATES.length);
  const tmpl = TEMPLATES[idx];

  console.log(`🎨 [BRAND] Template ${tmpl.id} | ${especialidadeId}`);

  // 1. Obter foto como buffer
  let fotoBuf = fotoBuffer || null;
  if (!fotoBuf && fotoUrl) {
    try {
      console.log(`📸 [BRAND] Buscando: ${fotoUrl.substring(0, 60)}...`);
      const r = await fetch(fotoUrl, { signal: AbortSignal.timeout(20000) });
      if (!r.ok) throw new Error(`HTTP ${r.status}`);
      fotoBuf = Buffer.from(await r.arrayBuffer());
      console.log(`📸 [BRAND] ${fotoBuf.length} bytes`);
    } catch (err) {
      console.warn(`⚠️ [BRAND] Foto falhou: ${err.message}`);
    }
  }

  // 2. Foto base 1080×1080 — crop pelo topo para preservar sujeitos
  let baseImg;
  if (fotoBuf) {
    baseImg = await sharp(fotoBuf)
      .resize(SIZE, SIZE, { fit: 'cover', position: 'north' })
      .jpeg({ quality: 92 })
      .toBuffer();
  } else {
    // Fallback: fundo verde claro se foto indisponível
    baseImg = await sharp({
      create: { width: SIZE, height: SIZE, channels: 3,
                background: { r: 220, g: 242, b: 232 } }
    }).jpeg().toBuffer();
  }

  // 3. Banda verde sólida no rodapé (sem transparência — cor pura)
  const band = await sharp({
    create: {
      width: SIZE, height: BAND_H, channels: 4,
      background: { ...BAND_COLOR, alpha: 255 }
    }
  }).png().toBuffer();

  const withBand = await sharp(baseImg)
    .composite([{ input: band, top: BAND_Y, left: 0 }])
    .jpeg({ quality: 92 })
    .toBuffer();

  // 4. Hook: primeira frase curta para subtítulo
  const primeiraFrase = (conteudo || titulo || '')
    .replace(/^.+\n/, '')               // remove primeira linha (é o título)
    .split(/[.!?]/)[0]
    .trim()
    .substring(0, 72);

  const hook = primeiraFrase || titulo?.substring(0, 72) || '';

  // 5. SVG com blobs + texto
  const label  = LABELS[especialidadeId] || 'FONO INOVA';
  const svgStr = gerarSVG({ tmpl, label, hook, bandY: BAND_Y });

  const imagemFinal = await sharp(withBand)
    .composite([{ input: Buffer.from(svgStr), blend: 'over' }])
    .jpeg({ quality: 92 })
    .toBuffer();

  // 6. Upload Cloudinary
  const base64 = `data:image/jpeg;base64,${imagemFinal.toString('base64')}`;
  const result = await cloudinary.uploader.upload(base64, {
    folder:    'fono-inova/branded',
    public_id: `${especialidadeId}_branded_${Date.now()}`,
  });

  console.log(`✅ [BRAND] ${result.secure_url}`);
  return result.secure_url;
}
